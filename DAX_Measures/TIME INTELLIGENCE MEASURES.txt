
// TIME INTELLIGENCE MEASURES

YTD Sales = 
TOTALYTD(
    [Total Sales],
    'Calendar Table'[Date]
)

MTD Sales = 
TOTALMTD(
    [Total Sales],
    'Calendar Table'[Date]
)

PTYD Sales = 
CALCULATE(
    [Total Sales],
    SAMEPERIODLASTYEAR('Calendar Table'[Date]),
    'Calendar Table'[Date] <= TODAY()
)
